Evaluate¶

, (Evaluation Metric) . Bike Sharing Demand Root Mean Squared Logarithmic Error (RMSLE) .

$$ \sqrt{\frac{1}{n} \sum_{i=1}^n (\log(p_i + 1) - \log(a_i+1))^2 } $$

($a_i$, actual) ($p_i$, predict) . (log(count + 1) ) numpy, scikit-learn RMSLE .

#   (linear algebra)  ,
#   '    '  .
import numpy as np

# scikit-learn  metrics  make_scorer   .
#       scikit-learn     .
from sklearn.metrics import make_scorer

# RMSLE    .
#   (predict) (actual)  .
def rmsle(predict, actual):
    # predict actual numpy array .
    #        .
    predict = np.array(predict)
    actual = np.array(actual)
    
    #   predict actual log(count + 1) .
    #    log_predict log_actual .
    log_predict = np.log(predict + 1)
    log_actual = np.log(actual + 1)
    
    #   log_predict log_actual   .
    #   distance    .
    distance = log_predict - log_actual
    
    #   distance .
    #   square_distance    .
    square_distance = distance ** 2
    
    #   square_distance  .
    #   mean_square_distance    .
    mean_square_distance = square_distance.mean()
    
    #   mean_square_distance (sqrt) .
    #   score    .
    score = np.sqrt(mean_square_distance)
    
    # score  .
    return score

# scikit-learn make_scorer 
# rmsle  scikit-learn      .
#   rmsle_score    .
rmsle_score = make_scorer(rmsle)
